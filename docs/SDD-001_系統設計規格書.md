## 1. 系統概述

### 1.1 系統目標
建立一個專注於包養和短期關係的安全、可擴展的交友平台，支援：
- 2,500+ 併發用戶
- 99.9% 系統可用性
- < 300ms API 響應時間
- 多重身份驗證和隱私保護

### 1.2 核心特性
- **匿名性保護**：用戶隱私優先的設計理念

- **即時通訊**：加密的實時聊天功能
- **智能配對**：基於偏好的推薦算法

### 1.3 技術目標
- **可擴展性**：支援用戶數量從百人到萬人的平滑擴展
- **高可用性**：系統故障時自動恢復，最小化服務中斷
- **安全性**：端對端加密，符合個資法規範

```php
class AuthenticationSystem {

        if ($user->hasMFAEnabled()) {
            return $this->requireMFAVerification($user);
        }
        
        // 3. 生成會話
        return $this->createUserSession($user);
    }
}
```

### 3.2 QR碼約會驗證系統

#### 3.2.1 核心實現

```php
// QR碼約會驗證服務
class DateVerificationService {
    
    public function createDateInvitation(CreateInvitationRequest $request): DateInvitation {
        $invitation = DateInvitation::create([
            'inviter_id' => $request->inviter_id,
            'invitee_id' => $request->invitee_id,
            'scheduled_at' => $request->scheduled_at,
            'location' => $request->location,
            'location_lat' => $request->location_lat,
            'location_lng' => $request->location_lng
        ]);
        
        // 生成QR碼內容（DB column = qr_token，PR-QR Step 2 規格錯誤矯正）
        $qrData = $this->generateQRData($invitation);
        $invitation->update(['qr_token' => $qrData]);
        
        // 發送通知
        $this->notificationService->sendDateInvitation($invitation);
        
        return $invitation;
    }
    
    private function generateQRData(DateInvitation $invitation): string {
        $payload = [
            'invitation_id' => $invitation->id,
            'inviter_id' => $invitation->inviter_id,
            'invitee_id' => $invitation->invitee_id,
            'scheduled_at' => $invitation->scheduled_at->timestamp,
            'expires_at' => $invitation->scheduled_at->addHours(1)->timestamp
        ];
        
        // 使用JWT加密QR碼內容
        return JWT::encode($payload, config('app.key'), 'HS256');
    }
    
    public function verifyDateAttendance(string $qrCode, int $userId, float $lat, float $lng): VerificationResult {
        try {
            // 解碼QR碼
            $payload = JWT::decode($qrCode, new Key(config('app.key'), 'HS256'));
            
            // 驗證邀請有效性
            $invitation = DateInvitation::findOrFail($payload->invitation_id);
            
            // 檢查時間窗口（約會時間前後30分鐘）
            $now = now();
            $scheduledTime = Carbon::createFromTimestamp($payload->scheduled_at);
            $timeWindow = 30; // 分鐘
            
            if ($now->lt($scheduledTime->subMinutes($timeWindow)) || 
                $now->gt($scheduledTime->addMinutes($timeWindow))) {
                throw new InvalidTimeWindowException();
            }
            
            // 檢查地理位置（500公尺範圍內）
            $distance = $this->calculateDistance(
                $lat, $lng,
                $invitation->location_lat,
                $invitation->location_lng
            );
            
            if ($distance > 500) { // 500公尺
                throw new LocationMismatchException();
            }
            
            // 記錄簽到
            $checkin = $this->recordCheckin($invitation, $userId, $lat, $lng);
            
            // 檢查是否雙方都已簽到
            if ($this->bothPartiesCheckedIn($invitation)) {
                $this->completeDateVerification($invitation);
            }
            
            return new VerificationResult(true, $checkin);
            
        } catch (Exception $e) {
            Log::error('Date verification failed', ['error' => $e->getMessage()]);
            return new VerificationResult(false, null, $e->getMessage());
        }
    }
}
```

#### 3.2.2 地理位置計算

```php
// 地理位置計算服務
class GeolocationService {
    
    /**
     * 計算兩點間距離（使用Haversine公式）
     */
    public function calculateDistance(float $lat1, float $lng1, float $lat2, float $lng2): float {
        $earthRadius = 6371000; // 地球半徑（公尺）
        
        $dLat = deg2rad($lat2 - $lat1);
        $dLng = deg2rad($lng2 - $lng1);
        
        $a = sin($dLat / 2) * sin($dLat / 2) +
             cos(deg2rad($lat1)) * cos(deg2rad($lat2)) *
             sin($dLng / 2) * sin($dLng / 2);
             
        $c = 2 * atan2(sqrt($a), sqrt(1 - $a));
        
        return $earthRadius * $c;
    }
    
    /**
     * 驗證地址是否在台灣境內
     */
    public function isLocationInTaiwan(float $lat, float $lng): bool {
        // 台灣邊界範圍
        $taiwanBounds = [
            'north' => 25.3,
            'south' => 21.9,
            'east' => 122.0,
            'west' => 119.3
        ];
        
        return $lat >= $taiwanBounds['south'] && 
               $lat <= $taiwanBounds['north'] &&
               $lng >= $taiwanBounds['west'] && 
               $lng <= $taiwanBounds['east'];
    }
}
```

```php
// WebSocket聊天服務器
class ChatServer {
    private $connections;
    private $redis;
    
    public function onOpen(ConnectionInterface $conn): void {
        // 用戶連接時驗證身份
        $user = $this->authenticateConnection($conn);
        
        if (!$user) {
            $conn->close();
            return;
        }
        
        // 存儲連接信息
        $this->connections[$user->id] = $conn;
        
        // 更新用戶在線狀態
        $this->updateOnlineStatus($user->id, true);
        
        Log::info("User {$user->id} connected to chat");
    }
    
    public function onMessage(ConnectionInterface $from, $msg): void {
        $data = json_decode($msg, true);
        $senderId = $this->getUserIdFromConnection($from);
        
        switch ($data['type']) {
            case 'private_message':
                $this->handlePrivateMessage($senderId, $data);
                break;
                
            case 'typing_indicator':
                $this->handleTypingIndicator($senderId, $data);
                break;
                
            case 'read_receipt':
                $this->handleReadReceipt($senderId, $data);
                break;
        }
    }
    
    private function handlePrivateMessage(int $senderId, array $data): void {
        $receiverId = $data['receiver_id'];
        $message = $data['message'];
        
        // 檢查用戶權限（誠信分數限制）
        if (!$this->canSendMessage($senderId, $receiverId)) {
            return;
        }
        
        // 加密消息內容
        $encryptedMessage = $this->encryptMessage($message);
        
        // 保存到數據庫
        $chatMessage = ChatMessage::create([
            'sender_id' => $senderId,
            'receiver_id' => $receiverId,
            'message' => $encryptedMessage,
            'message_type' => 'text',
            'sent_at' => now()
        ]);
        
        // 發送給接收者
        $this->sendToUser($receiverId, [
            'type' => 'new_message',
            'message' => [
                'id' => $chatMessage->id,
                'sender_id' => $senderId,
                'message' => $message,
                'sent_at' => $chatMessage->sent_at->toISOString()
            ]
        ]);
        
        // 如果用戶離線，發送推送通知
        if (!$this->isUserOnline($receiverId)) {
            $this->sendPushNotification($receiverId, $senderId, $message);
        }
    }
}
```

#### 3.3.2 消息加密系統

```php
// 消息加密服務
class MessageEncryptionService {
    private string $encryptionKey;
    
    public function __construct() {
        $this->encryptionKey = config('chat.encryption_key');
    }
    
    public function encryptMessage(string $message): string {
        $cipher = 'AES-256-GCM';
        $ivLength = openssl_cipher_iv_length($cipher);
        $iv = openssl_random_pseudo_bytes($ivLength);
        
        $encrypted = openssl_encrypt(
            $message,
            $cipher,
            $this->encryptionKey,
            OPENSSL_RAW_DATA,
            $iv,
            $tag
        );
        
        // 組合IV + Tag + 加密內容
        return base64_encode($iv . $tag . $encrypted);
    }
    
    public function decryptMessage(string $encryptedMessage): string {
        $data = base64_decode($encryptedMessage);
        $cipher = 'AES-256-GCM';
        $ivLength = openssl_cipher_iv_length($cipher);
        
        $iv = substr($data, 0, $ivLength);
        $tag = substr($data, $ivLength, 16);
        $encrypted = substr($data, $ivLength + 16);
        
        return openssl_decrypt(
            $encrypted,
            $cipher,
            $this->encryptionKey,
            OPENSSL_RAW_DATA,
            $iv,
            $tag
        );
    }
}
```

### 3.4 支付系統（v1.1 更新）

#### 3.4.1 支付網關整合

依據綠界官方最新技術文件直接串接，**完全不依賴第三方開源套件**：
- 金流 API：全方位金流 AioCheckOut/V5（[官方文件](https://developers.ecpay.com.tw/2509/)）
- 發票 API：B2C 電子發票 Issue（[官方文件](https://developers.ecpay.com.tw/7809/)）

```
付款流程：
  PaymentService::createOrder()
    → ECPayService::getPaymentUrl()
        ├─ sandbox → /mock?trade_no=… (開發用模擬付款)
        └─ production → 建立 AioCheckOut 表單
                        → cache 存 HTML
                        → 返回 /checkout/{token}
                        → 瀏覽器跳轉 ECPay 頁面

ECPay 回調：
  POST /payments/ecpay/notify
    → PaymentService::handleECPayNotify()
        ├─ 驗證 CheckMacValue (SHA256)
        ├─ 儲存對帳欄位 (TradeNo, PaymentDate, PaymentType, ChargeFee)
        ├─ 啟用訂閱 (activateSubscription)
        └─ 開立發票 (issueInvoiceForOrder)
             └─ ECPayService::issueInvoice()
                  ├─ JSON → urlencode → AES-128-CBC → Base64
                  ├─ POST 至 einvoice(-stage).ecpay.com.tw
                  ├─ 解密回應 → 取得 InvoiceNo / InvoiceDate / RandomNumber
                  └─ 儲存至 orders 表
```

**動態環境切換：** 金鑰與模式儲存於 `system_settings` 資料表，後台 UI（`/admin/settings` → 金流與發票 Tab）可即時切換 Sandbox / Production 環境，不需修改程式碼或重新部署。

**對帳機制：** 後台 `/admin/payments` 每筆訂單顯示綠界交易序號、付款方式、發票號碼，點擊「查看詳情」可查看完整對帳資訊（付款時間、發票開立時間等），供財會人員查核。

### 4.1 身份驗證與授權（v1.2 更新 — Multi-Guard 架構）

系統採用**前後台身份完全物理隔離（Multi-Guard Architecture）**：

```
┌─────────────────────────────────────────────────────────┐
│                    Laravel Sanctum                       │
│                                                         │
│  ┌─────────────────────┐   ┌─────────────────────────┐  │
│  │  Guard: sanctum     │   │  Guard: admin            │  │
│  │  Provider: users    │   │  Provider: admins        │  │
│  │  Model: User        │   │  Model: AdminUser        │  │
│  │  Table: users       │   │  Table: admin_users      │  │
│  │                     │   │  Token ability: ['admin'] │  │
│  │  → 前台 /api/v1/*   │   │  → 後台 /api/v1/admin/*  │  │
│  └─────────────────────┘   └─────────────────────────┘  │
│                                                         │
│  Token 互不相通：前台 Token 無法存取後台 API（401）       │
│                  後台 Token 無法存取前台 API（401）       │
└─────────────────────────────────────────────────────────┘
```

**管理員調整前台帳號：** 管理員若需使用前台功能，須以前台流程註冊獨立帳號。後台提供 `PATCH /admin/members/{id}/permissions` 端點，可直接設定該帳號的會員等級（Lv0-3）、誠信分數（0-100）與帳號狀態，所有調整寫入 `credit_score_histories` + `admin_operation_logs`。

```php
class CacheManager {
    
    public function getUserProfile(int $userId): array {
        // L1: 內存緩存 (APCu)
        $cacheKey = "user_profile:{$userId}";
        
        if (apcu_exists($cacheKey)) {
            return apcu_fetch($cacheKey);
        }
        
        // L2: Redis緩存
        $redisKey = "user:{$userId}:profile";
        $cachedData = Redis::get($redisKey);
        
        if ($cachedData) {
            $data = json_decode($cachedData, true);
            apcu_store($cacheKey, $data, 300); // 5分鐘
            return $data;
        }
        
        // L3: 數據庫查詢
        $user = User::with(['photos', 'preferences'])->findOrFail($userId);
        $data = $user->toArray();
        
        // 緩存到Redis (1小時)
        Redis::setex($redisKey, 3600, json_encode($data));
        
        // 緩存到內存 (5分鐘)
        apcu_store($cacheKey, $data, 300);
        
        return $data;
    }
    
    public function invalidateUserCache(int $userId): void {
        // 清除所有層級的緩存
        apcu_delete("user_profile:{$userId}");
        Redis::del("user:{$userId}:profile");
        Redis::del("user:{$userId}:matches");
        Redis::del("user:{$userId}:preferences");
    }
}
```

```javascript
// 圖片優化服務
class ImageOptimizationService {
    constructor() {
        this.compressionRatio = 0.8;
        this.maxWidth = 1200;
        this.maxHeight = 1200;
    }
    
    async optimizeImage(file) {
        return new Promise((resolve) => {
            const canvas = document.createElement('canvas');
            const ctx = canvas.getContext('2d');
            const img = new Image();
            
            img.onload = () => {
                // 計算新尺寸
                const { width, height } = this.calculateNewDimensions(
                    img.width, 
                    img.height
                );
                
                canvas.width = width;
                canvas.height = height;
                
                // 繪製優化後的圖片
                ctx.drawImage(img, 0, 0, width, height);
                
                // 轉換為Blob
                canvas.toBlob(resolve, 'image/jpeg', this.compressionRatio);
            };
            
            img.src = URL.createObjectURL(file);
        });
    }
    
    calculateNewDimensions(originalWidth, originalHeight) {
        let { width, height } = { width: originalWidth, height: originalHeight };
        
        if (width > this.maxWidth) {
            height = (height * this.maxWidth) / width;
            width = this.maxWidth;
        }
        
        if (height > this.maxHeight) {
            width = (width * this.maxHeight) / height;
            height = this.maxHeight;
        }
        
        return { width: Math.round(width), height: Math.round(height) };
    }
}
```

```php
class AdminPermissionMiddleware
{
    public function handle(Request $request, Closure $next, string $module, string $action): Response
    {
        $admin = $request->user();
        if (!$admin) {
            return response()->json(['success' => false, 'code' => 401, 'message' => '未登入'], 401);
        }

        $hasPermission = $admin->role->permissions()
            ->where('module', $module)

        if (!$hasPermission) {
            return response()->json(['success' => false, 'code' => 403, 'message' => '無此操作權限'], 403);
        }

        return $next($request);
    }
}
```

| 誠信分數 | 0 - 100 | 直接設定目標值，系統自動計算 delta |
| 會員等級 | Lv0 / Lv1 / Lv1.5 / Lv2 / Lv3 | 含女性照片驗證等級 |
| 帳號狀態 | active / suspended | 停權時自動記錄 `suspended_at` |

**稽核軌跡（Audit Trail）：**
- **誠信分數變動** → 透過 `CreditScoreService::adjust()` 自動寫入 `credit_score_histories`（type: `admin_set`，含 `operator_id`、`score_before`、`score_after`、`delta`）
- **所有操作** → 由 `LogAdminOperation` middleware 自動記錄至 `admin_operation_logs`（含 `admin_id`、`action`、`request_summary`、`ip_address`）

**使用場景：** 管理員以前台流程註冊獨立測試帳號後，可透過此功能快速將測試帳號設為付費會員（Lv3）或調整誠信分數，而不需經過實際付款或驗證流程。

#### 9.1.5 資料生命週期管理（v1.6 更新）

系統實作**兩階段銷毀機制**：

| 階段 | 觸發 | 行為 | 儲存位置 |
|------|------|------|---------|
| 軟刪除 / 收回 | 用戶操作 | 訊息：`is_recalled=true` + `recalled_at` 標記（content 保留於 DB）<br>照片：移至 quarantine/ | DB / `quarantine/` 目錄 |
| 物理銷毀 | 每日排程 `gdpr:process-deletions` | `Message::forceDelete()` 已超過 `data_retention_days` 的 recalled 訊息；`deleteDirectory()` 過期隔離目錄 | 永久移除 |

**收回訊息的可見度分級（Option C）：**
- 一般用戶（前台）：對方與自己皆看不到 recalled 訊息原文，顯示「（訊息已收回）」佔位符。
- 後台 `admin` / `cs`：顯示「[已收回]」佔位符（`is_content_visible: false`），search 端點排除 recalled 結果。
- 後台 `super_admin`：retention 期內可看原文（`is_content_visible: true`），search 端點包含 recalled 結果；每次查看寫入 `admin_operation_logs` 供稽核 / 法遵調閱使用。
- 超過 `data_retention_days` 後由排程 forceDelete，任何角色都看不到。

**媒體隔離區（Quarantine）：**
- 用戶刪除照片時，`GdprService::quarantineFile()` 將檔案從 `uploads/` 移至 `quarantine/{YYYY-MM-DD}/` 而非直接刪除
- 帳號去識別化時，自動將用戶所有上傳檔案批次移入隔離區
- 排程 Job 依 `data_retention_days`（後台 super_admin 可調，預設 180 天）清除過期隔離目錄

> **隱私權政策現有文案與實際 retention 設定的對齊狀況**：本任務不修改 `/privacy` 頁面文案。
> 兩者是否需同步調整，另開產品 + 法務議題評估。

```php
class TicketNumberGenerator
{
    public static function generate(): string
    {
        $date = now()->format('Ymd');
        $key  = "ticket_seq:{$date}";
```

### 9.4 誠信分數自動停權機制設計

#### 9.4.1 分數變更後置處理

```php
// Observer：監聽誠信分數變更事件
class CreditScoreObserver
{
    public function updated(CreditScoreHistory $history): void
    {
        $user = User::find($history->user_id);

        // 分數歸零：觸發自動停權
        if ($user->credit_score <= 0 && $user->status === 'active') {
            $user->update(['status' => 'auto_suspended']);
            // 限制功能（僅保留 /profile 和 /appeal）
            Cache::put("suspended_user:{$user->id}", true, now()->addYears(1));
            // 發送停權通知 Email
            Mail::to($user->email)->send(new AccountAutoSuspendedMail($user));
            Log::channel('admin')->info("Auto suspended user #{$user->id}, credit_score={$user->credit_score}");
        }

        // 分數恢復至 30 以上：自動解停（需管理員手動調整觸發此 Observer）
        if ($user->credit_score >= 30 && $user->status === 'auto_suspended') {
            $user->update(['status' => 'active']);
            Cache::forget("suspended_user:{$user->id}");
            Mail::to($user->email)->send(new AccountReactivatedMail($user));
        }
    }
}
```

```php
// 中介層：檢查用戶是否被停權
class CheckAccountStatus
{
    // 停權用戶允許訪問的路由白名單
    private const ALLOWED_ROUTES = [
        'profile.show',      // 個人資料頁（唯讀）
        'appeal.index',      // 申訴頁面入口
        'appeal.store',      // 提交申訴
        'auth.logout',       // 登出
    ];

    {
        $user = $request->user();
        if ($user && $user->status === 'auto_suspended') {
            if (!in_array($request->route()->getName(), self::ALLOWED_ROUTES)) {

                    'message' => '帳號已被停權，請至申訴頁面提交申訴', 'code' => 'account_suspended'], 403);
            }
        }
        return $next($request);
    }
}
```

```php
// 每個用戶在每個頻道有固定的匿名代號（同一頻道不換代號）
class AnonymousAliasService
{
    public function getAlias(int $userId, int $channelId): string
    {
        $cacheKey = "anon_alias:{$channelId}:{$userId}";
        return Cache::rememberForever($cacheKey, function () use ($userId, $channelId) {
            // 以 user_id + channel_id 的 Hash 決定代號，保持一致性
            $hash = crc32("{$userId}_{$channelId}");
            $number = abs($hash) % 9999 + 1;
            return "神秘訪客#{$number}";
        });
    }
}
```
