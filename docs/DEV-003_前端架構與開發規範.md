## 4. API 請求封裝

### 4.1 Axios 實例設定（`src/api/client.ts`）

```typescript
import axios from 'axios'
import { useAuthStore } from '@/stores/auth'
import router from '@/router'

export const apiClient = axios.create({
  baseURL: import.meta.env.VITE_API_BASE_URL,
  withCredentials: true,     // 攜帶 Sanctum Cookie
  headers: {
    'Content-Type': 'application/json',
    'Accept': 'application/json',
    'X-Requested-With': 'XMLHttpRequest',
  }
})

// Request 攔截器：附加 CSRF Token
apiClient.interceptors.request.use((config) => {
  const token = document.cookie
    .split('; ')
    .find(row => row.startsWith('XSRF-TOKEN='))
    ?.split('=')[1]
  if (token) {
    config.headers['X-XSRF-TOKEN'] = decodeURIComponent(token)
  }
  return config
})

// Response 攔截器：統一錯誤處理
apiClient.interceptors.response.use(
  (response) => response,
  async (error) => {
    if (error.response?.status === 401) {
      const authStore = useAuthStore()
      authStore.clearAuth()
      await router.push('/login')
    }
    if (error.response?.status === 403 && error.response?.data?.code === 'ACCOUNT_SUSPENDED') {
      await router.push('/suspended')
    }
    return Promise.reject(error)
  }
)
```

## 5. Pinia Store 設計

### 5.1 Auth Store（`src/stores/auth.ts`）

```typescript
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { loginApi, logoutApi, fetchMeApi } from '@/api/auth'
import type { AuthUser } from '@/types/user'

export const useAuthStore = defineStore('auth', () => {
  // State
  const user = ref<AuthUser | null>(null)
  const isLoading = ref(false)

  // Getters
  const isLoggedIn = computed(() => !!user.value)
  const isSuspended = computed(() => user.value?.status === 'suspended')
  const isPaidMember = computed(() => user.value?.membership_level === 3)
  const creditScore = computed(() => user.value?.credit_score ?? 0)

  // Actions
  async function login(email: string, password: string) {
    isLoading.value = true
    try {
      // 先取得 CSRF Cookie
      await fetch(`${import.meta.env.VITE_API_BASE_URL.replace('/api/v1', '')}/sanctum/csrf-cookie`,
        { credentials: 'include' })
      await loginApi({ email, password })
      await fetchCurrentUser()
    } finally {
      isLoading.value = false
    }
  }

  async function fetchCurrentUser() {
    user.value = await fetchMeApi()
  }

  function clearAuth() {
    user.value = null
  }

  async function logout() {
    await logoutApi()
    clearAuth()
  }

  return { user, isLoading, isLoggedIn, isSuspended, isPaidMember, creditScore,
           login, logout, fetchCurrentUser, clearAuth }
})
```

### 5.2 Chat Store（`src/stores/chat.ts`）

```typescript
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'

export const useChatStore = defineStore('chat', () => {
  // 各對話未讀數：Map<conversationId, unreadCount>
  const unreadCounts = ref<Map<number, number>>(new Map())

  // 總未讀數（badge 顯示）
  const totalUnread = computed(() =>
    [...unreadCounts.value.values()].reduce((sum, n) => sum + n, 0)
  )

  // 顯示用：最多 99+
  const unreadBadge = computed(() =>
    totalUnread.value > 99 ? '99+' : String(totalUnread.value || '')
  )

  function setUnread(conversationId: number, count: number) {
    unreadCounts.value.set(conversationId, count)
  }

  function clearUnread(conversationId: number) {
    unreadCounts.value.set(conversationId, 0)
  }

  return { unreadCounts, totalUnread, unreadBadge, setUnread, clearUnread }
})
```

## 6. WebSocket 整合

### 6.1 WebSocket Composable（`src/composables/useChat.ts`）

```typescript
import { io, Socket } from 'socket.io-client'
import { ref, onUnmounted } from 'vue'
import { useChatStore } from '@/stores/chat'
import { useAuthStore } from '@/stores/auth'

let socket: Socket | null = null

export function useChat() {
  const chatStore = useChatStore()
  const authStore = useAuthStore()
  const connected = ref(false)

  function connect() {
    if (socket?.connected) return

    socket = io(`${import.meta.env.VITE_WS_SCHEME || 'ws'}://${import.meta.env.VITE_WS_HOST}:${import.meta.env.VITE_WS_PORT}`, {
      auth: { key: import.meta.env.VITE_WS_KEY }
    })

    socket.on('connect', () => { connected.value = true })
    socket.on('disconnect', () => { connected.value = false })

    // 訂閱個人通知頻道
    const userId = authStore.user?.id
    if (userId) {
      socket.emit('subscribe', { channel: `private-notify.${userId}` })
      socket.on(`private-notify.${userId}`, handleNotification)
    }
  }

  function handleNotification(event: { type: string; data: any }) {
    if (event.type === 'new-message') {
      chatStore.setUnread(event.data.conversation_id,
        (chatStore.unreadCounts.get(event.data.conversation_id) ?? 0) + 1)
    }
  }

  function disconnect() {
    socket?.disconnect()
    socket = null
  }

  onUnmounted(() => {
    // 元件卸載時不斷開（保持全局連線），由 App.vue 管理
  })

  return { connected, connect, disconnect }
}
```

### 7.3 守衛實作（`src/router/index.ts`）

```typescript
import { createRouter, createWebHistory } from 'vue-router'
import { useAuthStore } from '@/stores/auth'

const router = createRouter({ history: createWebHistory(), routes })

router.beforeEach(async (to) => {
  const auth = useAuthStore()
  await auth.initialize() // 頁面重整後從 localStorage 恢復 token 並驗證

  const requiresAuth = to.meta.requiresAuth as boolean ?? false
  const minLevel     = (to.meta.minLevel as number) ?? 0

  // 未登入 → 導向登入頁
  if (requiresAuth && !auth.isLoggedIn) {
    return { path: '/login', query: { redirect: to.fullPath } }
  }

  // 等級不足 → 導向商城
  if (auth.isLoggedIn && auth.membershipLevel < minLevel) {
    return { path: '/app/shop', query: { reason: 'upgrade_required' } }
  }

  // 停權用戶
  if (auth.isLoggedIn && auth.user?.status === 'suspended') {
    if (!to.path.startsWith('/suspended')) return { path: '/suspended' }
  }

  // 帳號刪除冷靜期（pending_deletion）
  if (auth.isLoggedIn && auth.user?.status === 'pending_deletion') {
    if (to.path !== '/app/settings/delete-account') {
      return { path: '/app/settings/delete-account' }
    }
  }

  // 已登入不應訪問登入/註冊頁
  if (!requiresAuth && auth.isLoggedIn && ['/login', '/register'].includes(to.path)) {
    return { path: '/app/explore' }
  }

  return true
})
```

### 7.4 訪客頁點擊守衛（`src/composables/useVisitorClick.ts`）

SPEC-CONFIRM-001 B.7 確認：**查看訪客資料頁**對未付費男性限制，女性完全免費。此邏輯屬頁面內功能守衛，非路由層守衛。

```typescript
// src/composables/useVisitorClick.ts
import { useRouter } from 'vue-router'
import { useAuthStore } from '@/stores/auth'
import { useUpgradeModal } from '@/composables/useUpgradeModal'

export function useVisitorClick() {
  const router = useRouter()
  const auth = useAuthStore()
  const { showUpgradeModal } = useUpgradeModal()

  function handleVisitorClick(visitorUserId: number) {
    const user = auth.user
    if (!user) return

    if (user.gender === 'female') {
      // 女性：任何等級直接跳轉
      router.push(`/app/profiles/${visitorUserId}`)
      return
    }

    if (user.gender === 'male') {
      if (user.membership_level >= 3) {
        // 付費男性：直接跳轉
        router.push(`/app/profiles/${visitorUserId}`)
      } else {
        // 未付費男性：顯示升級提示
        showUpgradeModal('升級付費會員即可查看訪問過你的對象')
      }
    }
  }

  return { handleVisitorClick }
}
```

## 8. 誠信分數顯示規範

誠信分數為平台核心視覺元素，需統一顯示方式：

```typescript
// src/utils/creditScore.ts
export type CreditLevel = 'top' | 'good' | 'normal' | 'restricted'

export interface CreditDisplay {
  level: CreditLevel
  label: string
  color: string       // Tailwind text color class
  bgColor: string     // Tailwind bg color class
}

export function getCreditDisplay(score: number): CreditDisplay {
  if (score >= 91) return { level: 'top',        label: '頂級',   color: 'text-yellow-600',  bgColor: 'bg-yellow-50' }
  if (score >= 61) return { level: 'good',       label: '優質',   color: 'text-green-600',   bgColor: 'bg-green-50' }
  if (score >= 31) return { level: 'normal',     label: '普通',   color: 'text-blue-600',    bgColor: 'bg-blue-50' }
  return               { level: 'restricted',    label: '受限',   color: 'text-red-600',     bgColor: 'bg-red-50' }
}
```

## 9. 表單驗證規範

```typescript
// src/utils/validators.ts
export const rules = {
  required: (v: string) => !!v.trim() || '此欄位為必填',
  email: (v: string) => /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(v) || '請輸入有效的 Email',
  password: (v: string) => v.length >= 8 || '密碼至少需要 8 個字元',
  passwordConfirm: (v: string, original: string) => v === original || '兩次密碼不一致',
  phone: (v: string) => /^09\d{8}$/.test(v) || '請輸入台灣手機號碼（09xxxxxxxx）',
  nickname: (v: string) => (v.length >= 2 && v.length <= 20) || '暱稱需為 2-20 個字元',
}
```
